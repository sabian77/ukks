use App\Models\pkl;
Route::view('/pkl',  'pkl', ['pkl' => pkl::all()])->name('pkl'); 
Route:view untuk menampilkan file yg berada di resources/view
/pkl itu slug untuk web kita 
'pkl' adalaha file bladenya
